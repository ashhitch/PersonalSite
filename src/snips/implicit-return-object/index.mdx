### ES5

```js
const make = 'Ford';
const model = 'Mustang';
const engine = '5.0';

const car = {
  make: make,
  model: model,
  engine: engine,
};
```

### ES6

```js
const make = 'Ford';
const model = 'Mustang';
const engine = '5.0';

const car = {
  make,
  model,
  engine,
};
```
